// Example code to open a file in C

int main() {
    FILE *file;
    char filename[] = "example.txt";

    file = fopen(filename, "r");
    if (file == NULL) {
        printf("Error: Could not open file.\n");
        return 1;
    }

    printf("File opened successfully.\n");
    fclose(file);
    return 0;
}
